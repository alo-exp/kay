    use serde_json::Value;

    use crate::error::ToolError;

    pub struct FsReadTool {
        name: ToolName,
        description: String,
        input_schema: Value,
    }

    impl FsReadTool {
        pub fn new() -> Self {
            let name = ToolName::from("fs_read");
            let description = "Read a file from disk. For large files, use start_line/end_line range \
                reads rather than full reads."
                .to_string();
            let mut schema = serde_json::to_value(
                schemars::schema_for!(forge_domain::FSRead)
            ).unwrap_or_else(|_| serde_json::json!({ "type": "object" }));
            harden_tool_schema(
                &mut schema,
                &TruncationHints {
                    output_truncation_note: Some(
                        "For large files, use start_line/end_line range reads.".to_string(),
                    ),
                },
            );
            Self { name, description, input_schema: schema }
        }
    }

    #[async_trait]
    impl Tool for FsReadTool {
        fn name(&self) -> &ToolName { &self.name }
        fn description(&self) -> &str { &self.description }
        fn input_schema(&self) -> serde_json::Value { self.input_schema.clone() }

        async fn invoke(
            &self,
            args: Value,
            ctx: &ToolCallContext,
            _call_id: &str,
        ) -> Result<ToolOutput, ToolError> {
            let args = if args.is_null() { serde_json::json!({}) } else { args };
            let input: forge_domain::FSRead = serde_json::from_value(args)
                .map_err(|e| ToolError::InvalidArgs {
                    tool: self.name.clone(),
                    reason: e.to_string(),
                })?;

            // Delegate byte-for-byte. The forge-side ctx is constructed from ours.
            let forge_ctx = ctx.to_forge_tool_call_context();  // helper lives in kay-tools::tool.
            let forge_output = ctx.services
                .tool_executor()
                .execute(ToolCatalog::Read(input), &forge_ctx)
                .await
                .map_err(ToolError::from)?;  // ToolError: From<forge_app::ToolError>.

            Ok(ToolOutput::from_forge(forge_output))
        }
    }

    use crate::error::{CapScope, ToolError};
    use crate::tool::ToolName;

    pub struct ImageQuota {
        turn: AtomicU32,
        session: AtomicU32,
        max_per_turn: u32,
        max_per_session: u32,
    }

    impl ImageQuota {
        pub fn new(max_per_turn: u32, max_per_session: u32) -> Self {
            Self {
                turn: AtomicU32::new(0),
                session: AtomicU32::new(0),
                max_per_turn,
                max_per_session,
            }
        }

        pub fn try_consume(&self, scope: CapScope) -> Result<(), ToolError> {
            match scope {
                CapScope::Turn => Self::bump(&self.turn, self.max_per_turn, CapScope::Turn),
                CapScope::Session => Self::bump(&self.session, self.max_per_session, CapScope::Session),
            }
        }

        fn bump(
            counter: &AtomicU32,
            max: u32,
            scope: CapScope,
        ) -> Result<(), ToolError> {
            let prev = counter.fetch_add(1, Ordering::Relaxed);
            if prev + 1 > max {
                counter.fetch_sub(1, Ordering::Relaxed);
                return Err(ToolError::ImageCapExceeded {
                    tool: ToolName::from("image_read"),
                    scope,
                });
            }
            Ok(())
        }

        pub fn reset_turn(&self) {
            self.turn.store(0, Ordering::Relaxed);
        }

        #[cfg(test)]
        fn turn_count(&self) -> u32 { self.turn.load(Ordering::Relaxed) }
        #[cfg(test)]
        fn session_count(&self) -> u32 { self.session.load(Ordering::Relaxed) }
    }

    /// Detect MIME from file extension first, falling back to magic bytes.
    fn detect_mime(path: &str, bytes: &[u8]) -> &'static str {
        let lower = path.to_ascii_lowercase();
        if lower.ends_with(".png") { return "image/png"; }
        if lower.ends_with(".jpg") || lower.ends_with(".jpeg") { return "image/jpeg"; }
        if lower.ends_with(".gif") { return "image/gif"; }
        if lower.ends_with(".webp") { return "image/webp"; }
        // Magic-byte fallback.
        if bytes.starts_with(b"\x89PNG\r\n\x1a\n") { return "image/png"; }
        if bytes.starts_with(&[0xFF, 0xD8, 0xFF]) { return "image/jpeg"; }
        if bytes.starts_with(b"GIF87a") || bytes.starts_with(b"GIF89a") { return "image/gif"; }
        if bytes.len() >= 12 && &bytes[0..4] == b"RIFF" && &bytes[8..12] == b"WEBP" {
            return "image/webp";
        }
        "application/octet-stream"
    }

    impl Tool for ImageReadTool {
        fn name(&self) -> &ToolName { &self.name }
        fn description(&self) -> &str { &self.description }
        fn input_schema(&self) -> serde_json::Value { self.input_schema.clone() }

        async fn invoke(
            &self,
            args: Value,
            ctx: &ToolCallContext,
            _call_id: &str,
        ) -> Result<ToolOutput, ToolError> {
            let args = if args.is_null() { serde_json::json!({}) } else { args };
            let input: ImageReadArgs = serde_json::from_value(args)
                .map_err(|e| ToolError::InvalidArgs {
                    tool: self.name.clone(),
                    reason: e.to_string(),

            ctx.image_budget.try_consume(CapScope::Session)?;

            // Read via sandbox seam. Sandbox is responsible for path-traversal
            // guards (D-12). NoOpSandbox falls through to std::fs::read.
            let bytes = ctx.sandbox.read_bytes(&input.path).map_err(|e| ToolError::Input {
                tool: self.name.clone(),
                msg: format!("image read failed for {}: {}", input.path, e),
            })?;

            let size_bytes = bytes.len();
            let mime = detect_mime(&input.path, &bytes);
            let b64 = BASE64.encode(&bytes);
            let data_uri = format!("data:{};base64,{}", mime, b64);

            // Emit observability frame.
            let _ = ctx.events.send(AgentEvent::ImageRead {
                path: input.path.clone(),
                size_bytes,
                mime: mime.to_string(),
            }).await;

            Ok(ToolOutput::text(data_uri))
        }
    }
