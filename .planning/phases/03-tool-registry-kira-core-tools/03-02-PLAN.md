    #![allow(clippy::unwrap_used, clippy::expect_used)]

    use std::sync::Arc;

    use async_trait::async_trait;
    use forge_domain::{ToolCallContext, ToolName, ToolOutput};
    use kay_tools::{Tool, ToolError, ToolRegistry};
    use schemars::Schema;
    use serde_json::{json, Value};

    struct FakeTool {
        name: ToolName,
        description: String,
        schema: Schema,
    }

    #[async_trait]
    impl Tool for FakeTool {
        fn name(&self) -> &ToolName { &self.name }
        fn description(&self) -> &str { &self.description }
        fn input_schema(&self) -> &Schema { &self.schema }
        async fn invoke(
            &self,
            _args: Value,
            _ctx: &ToolCallContext,
            _call_id: &str,

            Ok(ToolOutput::text(format!("invoked {}", self.name)))

        let schema_val = json!({
            "type": "object",
            "properties": {"path": {"type": "string"}},
            "required": ["path"],
