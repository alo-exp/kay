    fn list(args: ListArgs) -> Result<()> {
        let store_dir = kay_session::kay_home().join("sessions");
        let store = kay_session::SessionStore::open(&store_dir)?;
        let sessions = kay_session::index::list_sessions(&store, args.limit)?;

        if sessions.is_empty() {
            println!("No sessions found");
            return Ok(());
        }

        if args.format == "json" {
            // Serialize to JSON array
            let json = serde_json::to_string_pretty(&sessions_to_json(&sessions))?;
            println!("{json}");
        } else {
            // Table format: header + rows
            println!("{:<36}  {:<8}  {:<20}  {}", "ID", "STATUS", "STARTED", "TITLE");
            for s in &sessions {
                println!("{:<36}  {:<8}  {:<20}  {}",
                    s.id, s.status, &s.start_time.format("%Y-%m-%d %H:%M:%S").to_string(),
                    &s.title);
            }
        }
        Ok(())
    }

            "title": s.title,

            "start_time": s.start_time.to_rfc3339(),
            "turn_count": s.turn_count,
            "cost_usd": s.cost_usd,
        })).collect::<Vec<_>>().into()

    pub fn rewind_cmd(args: RewindArgs) -> Result<()> {
        let store_dir = kay_session::kay_home().join("sessions");
        let store = kay_session::SessionStore::open(&store_dir)?;

        // Determine session_id: from --session flag or most recently active
        let session_id = match args.session {
            Some(id) => id,
            None => {
                // Find most recent active session
                let sessions = kay_session::index::list_sessions(&store, 1)?;
                sessions.into_iter().next()
                    .map(|s| s.id)
                    .ok_or_else(|| anyhow::anyhow!("no sessions found for rewind"))?
            }
        };

        // Get snapshots to restore
        let snap_paths = kay_session::snapshot::rewind(&store, &session_id)?;

        if snap_paths.is_empty() {
            anyhow::bail!("no snapshots available for session {session_id}");
        }

        // DL-8: --dry-run prints only, --force skips confirmation, else prompt
        if args.dry_run {
            println!("Would restore {} file(s):", snap_paths.len());
            for p in &snap_paths {
                println!("  {}", p.display());
            }
            return Ok(());
        }

        // Non-interactive without --force: ConfirmationRequired
        let is_tty = atty::is(atty::Stream::Stdin);
        if !is_tty && !args.force {
            return Err(kay_session::SessionError::ConfirmationRequired.into());
        }

        // Force or TTY: restore files
        // (TTY confirmation prompting deferred to Phase 10 UX polish;
        //  for now: --force skips, TTY assumes confirmed)
        // rewind() already computed the restored paths — files are restored there.
        // (rewind() in W-4 does the actual restore; this handler just calls it)
        println!("Restored {} file(s).", snap_paths.len());
        Ok(())
    }
