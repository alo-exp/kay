  ```bash
  SUBTREE=forge_app
  SUBDIR=crates/kay-core/src/$SUBTREE

  # Rule 1a: non-grouped intra-subtree imports
  files=$(grep -rlE '^use crate::' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe "s{^use crate::(?!forge_)(?!\{)}{use crate::$SUBTREE::}g"
  fi

  # Rule 1b: grouped intra-subtree imports
  files=$(grep -rlE '^use crate::\{' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe "s{^use crate::\{}{use crate::$SUBTREE::{}g"
  fi

  # Rule 2: inter-subtree imports (95 occurrences to upstream forge_config/display/domain/json_repair/stream/template)
  files=$(grep -rlE '^use forge_[a-z_]+::' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe 's{^use (forge_[a-z_]+::)}{use crate::$1}g'
  fi

  # Handle `pub use crate::X` and `pub use forge_X::` — same rules, `^pub use` prefix
  files=$(grep -rlE '^pub use crate::' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe "s{^pub use crate::(?!forge_)(?!\{)}{pub use crate::$SUBTREE::}g"
  fi
  files=$(grep -rlE '^pub use crate::\{' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe "s{^pub use crate::\{}{pub use crate::$SUBTREE::{}g"
  fi
  files=$(grep -rlE '^pub use forge_[a-z_]+::' "$SUBDIR" 2>/dev/null || true)
  if [ -n "$files" ]; then
    echo "$files" | xargs perl -i -pe 's{^pub use (forge_[a-z_]+::)}{pub use crate::$1}g'
  fi

  # Verify invariants
  remain=$(grep -rh "^\(pub \)\?use crate::" "$SUBDIR" 2>/dev/null \
    | grep -v "use crate::$SUBTREE" \
    | grep -v "use crate::forge_" \
    | wc -l | tr -d ' ')
  legacy=$(grep -rh "^\(pub \)\?use forge_[a-z_]\+::" "$SUBDIR" 2>/dev/null | wc -l | tr -d ' ')
  if [ "$remain" != "0" ] || [ "$legacy" != "0" ]; then
    echo "FAIL forge_app: remain=$remain legacy=$legacy (surfacing first 30)"
    grep -rh "^\(pub \)\?use crate::" "$SUBDIR" 2>/dev/null | grep -v "use crate::$SUBTREE" | grep -v "use crate::forge_" | head -15
    grep -rh "^\(pub \)\?use forge_[a-z_]\+::" "$SUBDIR" 2>/dev/null | head -15
    exit 1
  fi

  # Smoke check — forge_app should now compile clean, but upstream (forge_services/infra/repo/api/main)
  # are still broken, so the overall cargo check still fails. That is expected.
  cargo check -p kay-core 2>&1 | tail -10 || true

  git add -A crates/kay-core/src/forge_app
  git commit -s -m "refactor(02-04): path-rewrite forge_app (116 intra + 95 inter)

D-01 Step 2 per 02-CONTEXT.md. forge_app is the largest subtree (103 files)
and gets its own plan. Depends on 02-03 rewrites of forge_config, forge_display,
forge_domain, forge_json_repair, forge_stream, forge_template.

This also unlocks plan 02-08 (upstream delegation translator), which consumes
crate::forge_app::dto::openai::{Request, Response, ...}.

Co-Authored-By: Claude Opus 4.7 (1M context) <noreply@anthropic.com>"
  ```
