```rust
// crates/kay-provider-openrouter/src/cost_cap.rs

use std::sync::Mutex;

use crate::error::ProviderError;

pub struct CostCap {
    cap_usd: Option<f64>,
    spent_usd: Mutex<f64>,
}

impl CostCap {
    /// Uncapped session (D-10 default).
    pub fn uncapped() -> Self {
        Self { cap_usd: None, spent_usd: Mutex::new(0.0) }
    }

    /// Cap at the given USD amount. Must be > 0.
    pub fn with_cap(cap: f64) -> Result<Self, ProviderError> {
        if !(cap > 0.0) || !cap.is_finite() {
            return Err(ProviderError::Stream(
                "--max-usd must be a positive finite number".into(),
            ));
        }
        Ok(Self { cap_usd: Some(cap), spent_usd: Mutex::new(0.0) })
    }

    /// Called BEFORE fire of chat() request. Returns CostCapExceeded if
    /// accumulated spend already crosses the cap.
    pub fn check(&self) -> Result<(), ProviderError> {
        let Some(cap) = self.cap_usd else { return Ok(()); };
        // Lock poisoning in a cost-cap accumulator is recoverable — the
        // inner value is a simple f64; clear the poison and proceed.
        let spent = self.spent_usd.lock().unwrap_or_else(|e| e.into_inner());
        if *spent > cap {
            Err(ProviderError::CostCapExceeded { cap_usd: cap, spent_usd: *spent })
        } else {
            Ok(())
        }
    }

    /// Called on every AgentEvent::Usage arrival.
    pub fn accumulate(&self, cost_usd: f64) {
        let mut spent = self.spent_usd.lock().unwrap_or_else(|e| e.into_inner());
        *spent += cost_usd.max(0.0); // guard against negative cost noise
    }

    pub fn spent_usd(&self) -> f64 {
        *self.spent_usd.lock().unwrap_or_else(|e| e.into_inner())
    }

    pub fn cap_usd(&self) -> Option<f64> {
        self.cap_usd
    }
}
```

  ```rust
  //! Per-session cost cap (PROV-06, D-10). Accumulates USD spend from
  //! AgentEvent::Usage frames and rejects the NEXT turn if the accumulator
  //! exceeds the cap. Turn-boundary enforcement only — in-flight streaming
  //! is not aborted (Pitfall 3).
  //!
  //! `CostCap::uncapped()` is the default (no cap). `CostCap::with_cap(n)`
  //! requires n > 0 and finite; `--max-usd 0` becomes a usage error.

  use std::sync::Mutex;

  use crate::error::ProviderError;

  pub struct CostCap {
      cap_usd: Option<f64>,
      spent_usd: Mutex<f64>,
  }

  impl CostCap {
      /// Default uncapped session (D-10).
      pub fn uncapped() -> Self {
          Self {
              cap_usd: None,
              spent_usd: Mutex::new(0.0),
          }
      }

      /// Cap at `cap_usd`. Must be positive and finite; --max-usd 0 or a
      /// negative / NaN / infinite value becomes ProviderError::Stream.
      pub fn with_cap(cap: f64) -> Result<Self, ProviderError> {
          if !(cap > 0.0) || !cap.is_finite() {
              return Err(ProviderError::Stream(
                  "--max-usd must be a positive finite number".into(),
              ));
          }
          Ok(Self {
              cap_usd: Some(cap),
              spent_usd: Mutex::new(0.0),
          })
      }

      /// Pre-turn cost gate. Returns CostCapExceeded if accumulated spend is
      /// already past the cap. Called BEFORE any HTTP call.
      pub fn check(&self) -> Result<(), ProviderError> {
          let Some(cap) = self.cap_usd else { return Ok(()); };
          // NOTE (INFO #2 revision 2026-04-20): `.unwrap_or_else(|e| e.into_inner())`
          // is crate-level-lint-clean. `#![deny(clippy::unwrap_used)]` forbids
          // `.unwrap()` and `.expect()` but does NOT forbid `unwrap_or_else`,
          // which takes a closure and is a totally different API. Here the
          // closure recovers the f64 accumulator even if a prior holder panicked
          // while holding the lock (PoisonError -> into_inner). Since the
          // protected value is a plain f64 (not an invariant-carrying state
          // machine), clearing the poison and proceeding is the correct
          // recovery strategy.
          let spent = *self.spent_usd.lock().unwrap_or_else(|e| e.into_inner());
          if spent > cap {
              Err(ProviderError::CostCapExceeded {
                  cap_usd: cap,
                  spent_usd: spent,
              })
          } else {
              Ok(())
          }
      }

      /// Invoked on every AgentEvent::Usage arrival. Negative cost values
      /// (noise) are clamped to 0.
      pub fn accumulate(&self, cost_usd: f64) {
          // Same poisoning-is-recoverable rationale as `check()` above.
          // See that method's INFO #2 comment for the full explanation.
          let mut s = self.spent_usd.lock().unwrap_or_else(|e| e.into_inner());
          *s += cost_usd.max(0.0);
      }

      pub fn spent_usd(&self) -> f64 {
          *self.spent_usd.lock().unwrap_or_else(|e| e.into_inner())
      }

      pub fn cap_usd(&self) -> Option<f64> {
          self.cap_usd
      }
  }

  #[cfg(test)]
  mod unit {
      use super::*;

      #[test]
      fn uncapped_always_passes() {
          let c = CostCap::uncapped();
          c.accumulate(100_000.0);
          assert!(c.check().is_ok());
      }

      #[test]
  ```

  **Step 1 — Add the `retry_with_emitter` helper to `src/openrouter_provider.rs`:**

  ```rust
  use std::time::Duration;
  use bytes::Bytes;
  use futures::StreamExt;
  use reqwest_eventsource::EventSource;

  use crate::client::UpstreamClient;
  use crate::error::{ProviderError, RetryReason};
  use crate::event::AgentEvent;
  use crate::retry::{default_backoff, is_retryable};

  /// Executes `stream_chat` with retry (PROV-07). Returns the pre-collected
  /// `AgentEvent::Retry` frames together with the final Result. The caller
  /// yields the `Vec` first, then chains the translator stream.
  ///
  /// Contract (tested in `tests/retry_429_503.rs`):
  ///   - Retries fire only when `is_retryable(&err)` is true.
  ///   - `Retry-After` on 429 (`ProviderError::RateLimited { retry_after: Some(d) }`)
  ///     supersedes the backon schedule.
  ///   - Max 3 attempts; on exhaustion, the last `ProviderError` is surfaced.
  ///   - Every retry pushes `AgentEvent::Retry { attempt, delay_ms, reason }` BEFORE
  ///     `tokio::time::sleep(delay)`.
  async fn retry_with_emitter(
      upstream: &UpstreamClient,
      body: Bytes,
  ) -> (Vec<AgentEvent>, Result<EventSource, ProviderError>) {
      let mut pre_events: Vec<AgentEvent> = Vec::new();
      let mut schedule = default_backoff().build();
      let mut attempt: u32 = 0;
      loop {
          attempt += 1;
          match upstream.stream_chat(body.clone()).await {
              Ok(es) => return (pre_events, Ok(es)),
              Err(err) => {
                  if !is_retryable(&err) {
                      return (pre_events, Err(err));
                  }
                  let reason = match &err {
                      ProviderError::RateLimited { .. } => RetryReason::RateLimited,
                      // ServerError is the default retryable non-rate-limit classification.
                      // Network transport errors also map here per PATTERNS §RetryReason.
                      _ => RetryReason::ServerError,
                  };
                  // Retry-After overrides backon (D-09 + RESEARCH §Pitfall 6).
                  let delay = match &err {
                      ProviderError::RateLimited { retry_after: Some(d) } => *d,
                      _ => match schedule.next() {
                          Some(d) => d,
                          None => {
                              // Schedule exhausted — surface the error to the caller.
                              return (pre_events, Err(err));
                          }
                      },
                  };
                  // EMISSION: concrete AgentEvent::Retry construction at a named call site.
                  pre_events.push(AgentEvent::Retry {
                      attempt,
                      delay_ms: delay.as_millis() as u64,
                      reason,
                  });
                  tokio::time::sleep(delay).await;
              }
          }
      }
  }
  ```

  - `ProviderError::RateLimited { retry_after: Some(d) }` matches the variant shape

  ```rust
  /// Test-accessible generic retry loop. Production `retry_with_emitter`
  /// delegates here; tests use this directly with a stub closure.
  #[cfg(test)]
  async fn retry_with_emitter_using_unit_for_test<F, Fut, T>(
      mut fire: F,
  ) -> (Vec<AgentEvent>, Result<T, ProviderError>)
  where
      F: FnMut() -> Fut,
      Fut: std::future::Future<Output = Result<T, ProviderError>>,
  {
      use crate::retry::{default_backoff, is_retryable};
      let mut pre_events: Vec<AgentEvent> = Vec::new();
      let mut schedule = default_backoff().build();
      let mut attempt: u32 = 0;
      loop {
          attempt += 1;
          match fire().await {
              Ok(ok) => return (pre_events, Ok(ok)),
              Err(err) => {
                  if !is_retryable(&err) {
                      return (pre_events, Err(err));
                  }
                  let reason = match &err {
                      ProviderError::RateLimited { .. } => RetryReason::RateLimited,
                      _ => RetryReason::ServerError,
                  };
                  let delay = match &err {
                      ProviderError::RateLimited { retry_after: Some(d) } => *d,
                      _ => match schedule.next() {
                          Some(d) => d,
                          None => return (pre_events, Err(err)),
                      },
                  };
                  pre_events.push(AgentEvent::Retry {
                      attempt,
                      delay_ms: delay.as_millis() as u64,
                      reason,
                  });
                  tokio::time::sleep(delay).await;
              }
          }
      }
  }
  ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Retry loop → outer caller | Retry emissions (AgentEvent::Retry) are visible to the caller — UI can show "retrying". User can Ctrl-C to drop the stream (Pitfall 9). |
| Cost cap accumulator → concurrent sessions | Arc<CostCap> with Mutex<f64>; thread-safe. Concurrent chat() calls serialize on the cap check + accumulator update. |
| HTTP classifier → adversary-controlled response bodies | `classify_http_error` returns the response body verbatim in `ProviderError::Http { status, body }`. Caller must not log this unbounded — Phase 5 adds bounded logging. |

|-----------|----------|-----------|-------------|-----------------|
| TM-03 / T-02-10-01 | Repudiation / Self-DoS | Cost-cap bypass via concurrent requests | mitigate | Single Mutex<f64> in CostCap; concurrent `chat()` calls serialize on `check()` + `accumulate()`. Test `cost_cap_turn_boundary` asserts turn N+1 catches; concurrent case is a future hardening (documented as Phase 3+ concern per RESEARCH §TM-03). |
