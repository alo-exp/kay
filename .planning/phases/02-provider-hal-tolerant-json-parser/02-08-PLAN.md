// crates/kay-provider-openrouter/src/translator.rs (skeleton)

use std::collections::HashMap;

use futures::{Stream, StreamExt, stream};
use reqwest_eventsource::{Event, EventSource};

use crate::error::ProviderError;
use crate::event::AgentEvent;

struct ToolCallBuilder {
    name: Option<String>,
    arguments_raw: String,
}

impl ToolCallBuilder {
    fn new() -> Self { Self { name: None, arguments_raw: String::new() } }
}

pub(crate) fn translate_stream(
    mut source: EventSource,
) -> impl Stream<Item = Result<AgentEvent, ProviderError>> + Send + 'static {
    async_stream::stream! {
        // or hand-rolled with unfold() — either works; pick whatever keeps
        // the closure shape readable. `async_stream` is added as a dep below
        // per RESEARCH line 408 (planner-discretion).
        let mut builders: HashMap<String, ToolCallBuilder> = HashMap::new();
        while let Some(next) = source.next().await {
            match next {
                Ok(Event::Open) => {} // connection established; ignore
                Ok(Event::Message(ev)) if ev.data == "[DONE]" || ev.data.is_empty() => {
                    // Terminal sentinel. Flush any remaining open tool_calls
                    // as Complete (best-effort) — though OpenRouter sends
                    // finish_reason BEFORE [DONE], so builders is usually empty.
                    for (id, b) in builders.drain() {
                        yield try_complete(&id, b).into_event();
                    }
                    break;
                }
                Ok(Event::Message(ev)) => {
                    match serde_json::from_str::<kay_core::forge_app::dto::openai::Response>(&ev.data) {
                        Ok(resp) => {
                            // Walk resp.choices[0].delta, yielding events.
                            for event in events_from_response(resp, &mut builders) {
                                yield event;
                            }
                        }
                        Err(e) => {
                            yield Err(ProviderError::Stream(format!("parse chunk: {e}")));
                            break;
                        }
                    }
                }
                Err(reqwest_eventsource::Error::StreamEnded) => break,
                Err(e) => {
                    yield Err(upstream_error_to_provider(e).await);
                    break;
                }
            }
        }
    }
}

fn events_from_response(
    resp: kay_core::forge_app::dto::openai::Response,
    builders: &mut HashMap<String, ToolCallBuilder>,
) -> Vec<Result<AgentEvent, ProviderError>> {
    let mut out = Vec::new();
    // Examine resp.choices[0].delta:
    //   - If delta.content -> AgentEvent::TextDelta
    //   - If delta.tool_calls present, for each:
    //       - If id is new to builders -> builders.insert + emit ToolCallStart
    //       - If arguments string is non-empty -> append to builder + emit ToolCallDelta
    //   - If finish_reason is Some("tool_calls") or Some("stop"):
    //       drain builders; strict-parse builder.arguments_raw; emit Complete
    // - If resp.usage present -> emit AgentEvent::Usage
    //
    // Response DTO field reference (per WARNING #2 revision 2026-04-20):
    //   Post-rename path: `crates/kay-core/src/forge_app/dto/openai/response.rs`.
    //   Expected fields (inspect + adapt at impl time):
    //     Response { choices: Vec<Choice>, usage: Option<Usage> }
    //     Choice   { delta: Delta, finish_reason: Option<String | FinishReason> }
    //     Delta    { content: Option<String>, tool_calls: Option<Vec<ToolCallDelta>> }
    //     Usage    { prompt_tokens: u64, completion_tokens: u64, total_tokens: u64, cost: Option<f64> }
    //     ToolCallDelta { id: Option<String>, index: Option<u32>, function: Option<FunctionDelta> }
    //     FunctionDelta { name: Option<String>, arguments: Option<String> }
    //
    // STRICT parse (plan 02-09 will insert the tolerant 2nd pass here).
    // A strict-parse failure in this plan surfaces as:
    //   AgentEvent::Error { error: ProviderError::ToolCallMalformed { .. } }
    // AND TERMINATES THE STREAM.
    // Plan 02-09 will replace that behavior with emitting ToolCallMalformed
    // (an AgentEvent variant) instead of Error, so the stream can continue.
    //
    // If the DTO types are not shaped as described, adjust — the KEY INVARIANTS
    // (per-call_id reassembly, terminal-marker complete emission, strict pass
    // in this plan, Usage emission) are the CONTRACT, not the field-access syntax.

    out
}

async fn upstream_error_to_provider(e: reqwest_eventsource::Error) -> ProviderError {
    // Minimal mapping for Phase 2 plan 02-08 (streaming_happy_path.rs success path).
    // Full error taxonomy (401/402/429/5xx/transport) lands in plan 02-10 via
    // the `classify_http_error` helper. For this plan we handle the common
    // branches; plan 02-10 replaces this function entirely with a call to
    // `classify_http_error(status, headers, body)`.
    match e {
        reqwest_eventsource::Error::StreamEnded => ProviderError::Stream("stream ended".into()),
        reqwest_eventsource::Error::InvalidStatusCode(status, response) => ProviderError::Http {
            status: status.as_u16(),
            body: response.text().await.unwrap_or_else(|_| String::new()),
        },
        reqwest_eventsource::Error::InvalidContentType(_, response) => ProviderError::Http {
            status: response.status().as_u16(),
            body: "invalid content type".into(),
        },
        reqwest_eventsource::Error::Transport(err) => ProviderError::Network(err),
        other => ProviderError::Stream(format!("upstream: {other}")),
    }
}

// crates/kay-provider-openrouter/src/client.rs (skeleton)

use bytes::Bytes;
use reqwest::header::HeaderMap;
use reqwest::{Client, Url};
use reqwest_eventsource::{EventSource, RequestBuilderExt};

use crate::auth::ApiKey;
use crate::error::ProviderError;

pub(crate) struct UpstreamClient {
    client: Client,
    endpoint: Url,
    api_key: ApiKey,
    // Optional identification headers (OpenRouter recommends these; see openai.rs:51)
    referer: Option<String>,
    title: Option<String>,
}

impl UpstreamClient {
    /// Fallible ctor per revision 2026-04-20 (checker BLOCKER #3):
    /// `.expect()` on `Client::builder().build()` violates the crate-level
    /// `#![deny(clippy::expect_used)]` lint. `try_new` returns
    /// `Result<Self, ProviderError>` and maps reqwest build errors via
    /// `ProviderError::Network`. This is the canonical constructor — NO
    /// infallible `new()` variant exists.
    pub(crate) fn try_new(endpoint: Url, api_key: ApiKey) -> Result<Self, ProviderError> {
        let client = Client::builder()
            .timeout(std::time::Duration::from_secs(120))
            .build()
            .map_err(ProviderError::Network)?;
        Ok(Self { client, endpoint, api_key, referer: None, title: None })
    }

    pub(crate) fn with_headers(mut self, referer: Option<String>, title: Option<String>) -> Self {
        self.referer = referer;
        self.title = title;
        self
    }

    /// POST a JSON body to the endpoint and return an SSE EventSource.
    /// 401 / 402 / 429 / 5xx errors surface as ProviderError before the
    /// EventSource even opens; transport errors surface as Network.
    pub(crate) async fn stream_chat(&self, body: Bytes) -> Result<EventSource, ProviderError> {
        let mut headers = HeaderMap::new();
        // Authorization: Bearer <key>
        headers.insert(
            reqwest::header::AUTHORIZATION,
            format!("Bearer {}", self.api_key.as_str())
                .parse()
                .map_err(|_| ProviderError::Auth {
                    reason: crate::error::AuthErrorKind::Invalid,
                })?,
        );
        headers.insert(
            reqwest::header::CONTENT_TYPE,
            "application/json".parse().unwrap(), // static ASCII — cannot fail
        );
        // HTTP-Referer (OpenRouter docs, forge_repo/provider/openai.rs:51)
        if let Some(ref r) = self.referer {
            if let Ok(v) = r.parse() {
                headers.insert(reqwest::header::HeaderName::from_static("http-referer"), v);
            }
        }
        if let Some(ref t) = self.title {
            if let Ok(v) = t.parse() {
                headers.insert(reqwest::header::HeaderName::from_static("x-title"), v);
            }
        }

        let req = self.client
            .post(self.endpoint.clone())
            .headers(headers)
            .body(body);

        // `RequestBuilderExt::eventsource` returns EventSource (Err only on
        // misconfigured request — we have full control, so this is infallible
        // in practice; still propagate as Network for belt-and-braces).
        let es = req
            .eventsource()
            .map_err(|e| ProviderError::Stream(format!("eventsource: {e}")))?;
        Ok(es)
    }
}

        self

    pub fn build(self) -> Result<OpenRouterProvider, ProviderError> {
        let endpoint = self.endpoint
            .ok_or(ProviderError::Stream("endpoint required".into()))?;
        let endpoint_url = Url::parse(&endpoint)
            .map_err(|e| ProviderError::Stream(format!("bad endpoint url: {e}")))?;
        let allowlist = self.allowlist.unwrap_or_else(|| Allowlist::from_models(Vec::new()));
        let api_key = match self.api_key {
            Some(k) => k,
            None => resolve_api_key(self.config_auth.as_ref())?,
        };
        let upstream = UpstreamClient::try_new(endpoint_url, api_key)?
            .with_headers(self.referer, self.title);

            None => Arc::new(CostCap::uncapped()),

        };
        Ok(OpenRouterProvider { allowlist, upstream, cost_cap })
    }

#[async_trait]
impl Provider for OpenRouterProvider {
    async fn chat<'a>(
        &'a self,
        request: ChatRequest,
    ) -> Result<AgentEventStream<'a>, ProviderError> {
        // Pre-flight gate: allowlist
        self.allowlist.check(&request.model)?;

        // Rewrite to wire form (Pitfall 8, TM-08)
        let wire_model = self.allowlist.to_wire_model(&request.model);

        // Construct request body via forge_app::dto::openai::Request
        let body = build_request_body(&wire_model, &request)?;
        let bytes = Bytes::from(body);

        // Fire HTTP + get EventSource
        let es = self.upstream.stream_chat(bytes).await?;

        // Translate SSE stream -> AgentEvent stream
        let stream = translate_stream(es);
        Ok(Box::pin(stream))
    }

    async fn models(&self) -> Result<Vec<String>, ProviderError> {
        Ok(self.allowlist.models().to_vec())
    }
}

fn build_request_body(
    wire_model: &str,
    req: &ChatRequest,
) -> Result<Vec<u8>, ProviderError> {
    // BLOCKER #2 resolved (revision 2026-04-20): COMMITTED to Option A —
    // construct `kay_core::forge_app::dto::openai::Request` from our
    // `ChatRequest`. Rationale: EVAL-01 parity priority + ForgeCode reuse
    // (D-02). The ForgeCode DTO already encodes OpenRouter's wire format;
    // mapping our minimal ChatRequest into it preserves behavior parity.
    //
    // Field mapping (authoritative per this plan):
    //   - Our ChatRequest.model  -> Request.model        (with :exacto suffix
    //                                                     already applied by
    //                                                     caller via
    //                                                     allowlist.to_wire_model())
    //   - Our ChatRequest.messages -> Request.messages
    //       Each `Message { role, content, tool_call_id }` maps to
    //       `forge_app::dto::openai::Message` with the role enum variant
    //       resolved from the string ("system"/"user"/"assistant"/"tool").
    //   - Our ChatRequest.tools -> Request.tools
    //       Each `ToolSchema { name, description, input_schema }` maps to
    //       `forge_app::dto::openai::Tool` (function-type) and is fed through
    //       the `normalize_tool_schema` transformer per NN-7.
    //   - Our ChatRequest.temperature -> Request.temperature (Option<f32>)
    //   - Our ChatRequest.max_tokens  -> Request.max_tokens (Option<u32>)
    //   - stream: true  (SSE — always for Phase 2)
    //
    // The concrete construction below uses a `From<ChatRequest>` or direct
    // field-by-field initialisation (executor picks based on whether the DTO
    // derives Default, Setters, or a builder — inspect at impl time). The
    // hand-rolled JSON form (previously Option B) is NOT acceptable as final
    // code: it would diverge silently from ForgeCode's schema hardening.
    //
    // Implementation skeleton (executor completes with actual field names
    // post-rename inspection):
    use kay_core::forge_app::dto::openai::{Request, Message as OaiMessage, Role, Tool, Function};

    let oai_messages: Vec<OaiMessage> = req.messages.iter().map(|m| {
        let role = match m.role.as_str() {
            "system" => Role::System,
            "user" => Role::User,
            "assistant" => Role::Assistant,
            "tool" => Role::Tool,
            _ => Role::User, // defensive: unknown role falls through to user
        };
        let mut om = OaiMessage::default();
        om.role = role;
        om.content = Some(m.content.clone().into()); // adjust field shape post-rename inspection
        om.tool_call_id = m.tool_call_id.clone();
        om
    }).collect();

    let oai_tools: Vec<Tool> = req.tools.iter().map(|t| Tool::function(Function {
        name: t.name.clone(),
        description: Some(t.description.clone()),
        parameters: Some(t.input_schema.clone()),
        // strict: true — per NN-7 ForgeCode's normalize_tool_schema pipeline
        // will further harden this during serialize; we set the flag for
        // OpenRouter Exacto routing.

  //!     terminates the stream.

  use futures::StreamExt;

  use serde_json::Value;

  use crate::event::AgentEvent;

      name: Option<String>,
      arguments_raw: String,
  }

  impl ToolCallBuilder {

      if raw.is_empty() {

  pub(crate) fn translate_stream(
      mut source: EventSource,
  ) -> impl Stream<Item = Result<AgentEvent, ProviderError>> + Send + 'static {
      stream! {
          let mut builders: HashMap<String, ToolCallBuilder> = HashMap::new();
          let mut stream_ended_normally = false;

          while let Some(next) = source.next().await {
              match next {
                  Ok(Event::Open) => {
                      // Connection established; no event to emit.
                  }
                  Ok(Event::Message(ev)) => {
                      let data = ev.data.trim();
                      // Sentinel for stream end — OpenRouter convention per
                      // forge_repo/provider/event.rs line 26.
                      if data == "[DONE]" || data.is_empty() {
                          stream_ended_normally = true;
                          break;

                      // Parse chunk as forge_app::dto::openai::Response.
                      let resp: kay_core::forge_app::dto::openai::Response =
                          match serde_json::from_str(data) {
                              Ok(r) => r,
                              Err(e) => {
                                  yield Err(ProviderError::Stream(format!(
                                      "parse chunk failed: {e}"
                                  )));
                                  return;
                              }
                          };

                      // Walk choices[0].delta for text + tool_calls
                      for ev in events_from_response(&resp, &mut builders) {
                          yield ev;
                          // If we yielded a ToolCallMalformed variant via the
                          // strict parse failure path, the inner error case
                          // already broke out; no additional guard needed.
                      }
                  }
                  Err(reqwest_eventsource::Error::StreamEnded) => {
                      stream_ended_normally = true;
                      break;
                  }
                  Err(err) => {
                      // Terminal upstream error — map to ProviderError.
                      // Plan 02-10 fills in the detailed mapping; for now,
                      // surface as Network or Stream depending on variant.
                      yield Err(map_upstream_error(err).await);
                      return;
                  }
              }

          // (unusual — OpenRouter sends finish_reason before [DONE]),
          // flush them as Complete using the strict parse. Unparseable
          // arguments surface as Error in this plan; 02-09 upgrades.

                          });
